01.数组中重复的数字

利用count数组存0~n-1出现的次数, 当其>1时说明是重复元素

class Solution {
public int findRepeatNumber(int[] nums){
int[] count = new int[nums.length];
for(int n:nums){
count[n]++;
if(count[n]>1) return n;
}

    return -1;

}

}

02.二维数组中的查找

查找是否包含该数, 从右上角开始搜索

class Solution {
public boolean findNumberIn2DArray(int[][] matrix, int target) {
// 判空
if(matrix==null||matrix.length==0||matrix[0].length==0) return false;

    int m = 0;
    int n = matrix[0].length-1;
    while(m<matrix.length&&n>=0){
        // 如果==, 返回
        if(matrix[m][n]==target){
            return true;
        }
        // 如果<target, 说明第m行都比target小
        else if(matrix[m][n]<target){
            m++;
        }
        // 如果>target, 说明第n列都比target大
        else{
            n--;
        }
    }
    return false;
}

}

05.重建二叉树

给出前序,中序, 重建树
递归, 按照前序顺序重建节点 dfs(pre,in,null)
递归终止条件: 按前序已建立好所有节点,或者到了中序的截断点

class Solution {
// 前序
int preindex;
// 中序
int inindex;
public TreeNode buildTree(int[] preorder, int[] inorder) {
preindex = 0;
inindex = 0;
return dfs(preorder,inorder,null);
}
// pre:前序 in:中序 finish:分段截断点
// 功能：重建finish节点前的所有节点
TreeNode dfs(int[] pre,int[] in,TreeNode finish){
// 递归终止条件
// 如果按照前序遍历已经重建好了所有节点，或者到了中序里的截断点
if(preindex==pre.length||(finish!=null&&finish.val==in[inindex])) return null;

    // 重建根节点: 前序遍历
    TreeNode r = new TreeNode(pre[preindex]);
    preindex++;

    // 左 递归 中序遍历r之前的节点
    r.left = dfs(pre,in,r);
    inindex++;

    // 右
    r.right = dfs(pre,in,finish);

    return r;
}

}

06.用两个栈实现队列

class CQueue {
Stack A;
Stack B;

public CQueue() {
    A = new Stack<>();
    B = new Stack<>();
}

public void appendTail(int value) {
    // 添加时直接添加
    A.push(value);
}

public int deleteHead() {
    // 删除时倒到B中删除，B作为一个缓冲区使用
    // 如果B中此时有值，直接从B中返回
    if(!B.isEmpty()) return B.pop();
    else{
        while(!A.isEmpty()){
            B.push(A.pop());
        }
        if(!B.isEmpty()) return B.pop();
    }

    return -1;
}

}

09.矩阵中的路径

深度优先二维搜索 dfs(board,word,i,j,0)

class Solution {
int m;
int n;
public boolean exist(char[][] board, String word) {
//深度优先
if(board==null||board.length==0||board[0].length==0) return false;
m=board.length;
n=board[0].length;
for(int i=0;i<m;i++){
for(int j=0;j<n;j++){
if(dfs(board,word,i,j,0)) return true;
}
}
return false;
}
//b:字母矩阵 word:单词 i,j 起点 s:单词检索索引
boolean dfs(char[][] b,String word,int i,int j,int s){
//递归终止条件

    //越界或者不等 返回false
    if(i<0||i>=m||j<0||j>=n||b[i][j]!=word.charAt(s)) return false;
    //如果上面没有返回false且word全走完了 返回true
    if(s==word.length()-1) return true;
    char temp = b[i][j];
    //利用'*'代替visited标记
    b[i][j] = '*';
    //搜索上下左右
    boolean res = dfs(b,word,i+1,j,s+1)||dfs(b,word,i-1,j,s+1)||dfs(b,word,i,j+1,s+1)||dfs(b,word,i,j-1,s+1);

    //回溯
    b[i][j] = temp;

    return res;
}

}

visited[x][y] = true;
//19->20 和小了8
//18->19 和大了1
return 1+ dfs(x+1,y,(x+1)%10==0?xh-8:xh+1,yh)+dfs(x,y+1,xh,(y+1)%10==0?yh-8:yh+1);
}

class Solution {
public int cuttingRope(int n) {
//动态规划
//dp[i]=max(dp[i],dp[i-j]j,(i-j)j)
if(n<1) return 0;
//1. 初始化dp
int[] dp = new int[n+1];
dp[1] = 1;
//2. 遍历更新
for(int i=2;i<=n;i++){
for(int j=1;j<i;j++){
dp[i] = Math.max(Math.max(dp[i-j]j,(i-j)j),dp[i]);
}
}
return dp[n];
}
}

12.剪绳子(取模)

贪心

class Solution {
public int cuttingRope(int n) {
//需要取模的话就不能用动态规划了
//贪心算法
//尽可能地取3，最后余数可能是0，1，2
//0：结果即为所有三相乘
//1：取出一个三，22>31
//2: 所有三相乘再*2
if(n<1) return 0;
if(n==1) return 1;
if(n==2) return 1;
if(n==3) return 2;

    int e=1000000007;
    //res一定要用long类型,不然大数会出错
    long res = 1;   
    if(n%3==2) {
        n-=2;
        res=2;
    }
    else if(n%3==1){
        n-=4;
        res = 4;
    }

    while(n>0){
        res*=3;
        res = res%e;
        n-=3;
    }
    return (int)res%e;
}

}

14.数值的整数次方

class Solution {
public double myPow(double x, int n) {
//快速幂写法
if(n==0&&x==0) return -1;
else if(n==0) return 1;
else if(x==0) return 0;
//用long接收，防止越界
long p = n;

    if(p<0){
        x = 1/x;
        p = -p;
    }    
    double res = 1.0;
    while(p>0){
        if((p&1)==1) res*=x;
        x *=x;
        p >>=1;

    }
    return res;
}

}

17.正则式匹配

动态规划

class Solution {
public boolean isMatch(String s, String p) {
//动态规划
//dp[i][j]: 表示s的前i个和p的前j个相匹配，即0i-1和0j-1匹配
int m = s.length();
int n = p.length();
boolean[][] dp = new boolean[m+1][n+1];

    for(int i=0;i<=m;i++){
        for(int j=0;j<=n;j++){
            //初始化 dp[0][0]=true,其余dp[i][0]=false;
            if(j==0){
                dp[i][0] = (i==0);

            }else{
                //如果不是*
                if(p.charAt(j-1)!='*'){
                    //如果s[i-1]和p[i-1]相匹配，则前i个与前j个匹配情况同上一个相同
                    if(i>0&&(s.charAt(i-1)==p.charAt(j-1)||p.charAt(j-1)=='.')){
                        dp[i][j] = dp[i-1][j-1];
                    }
                }
                //如果是'*'
                else{
                    //'*'若取零,则前ij个的匹配情况就和前i j-2个的匹配相同
                    if(j>=2){

                        dp[i][j] |= dp[i][j-2];
                    }
                    //'*'不取零,就要看前i j-1个是否相匹配
                    //s[i-1]和p[j-2]比较，相等则匹配情况与前i-1 j个相同，i继续往后匹配
                    if(j>=2&&i>0&&(s.charAt(i-1)==p.charAt(j-2)||p.charAt(j-2)=='.')){
                        dp[i][j] |= dp[i-1][j];
                    }
                }
            }
        }
    }
    return dp[m][n];
}

}

18.表示数值的字符串

class Solution {
public boolean isNumber(String s) {
char[] ch = s.trim().toCharArray();
if(ch==null||ch.length==0) return false;
//设置标志位记录前面是否出现过该元素
boolean e = false;
boolean dot = false;
boolean num = false;

    for(int i=0;i<ch.length;i++){
        if(ch[i]>='0'&&ch[i]<='9'){
            num=true;
        }
        else if(ch[i]=='e'||ch[i]=='E'){
            //如果是e, e前面一定有数字,不能有e,且e之后一定得有数字
            if(!num||e) return false;
            e = true;
            //重置num,便于判断后面有没有数字
            num = false;
        }
        else if(ch[i]=='.'){
            //.前面不能有e,不能有.
            if(dot||e) return false;
            dot=true;
        }
        else if(ch[i]=='+'||ch[i]=='-'){
            //符号出现在第一位或者e后面一位,否则不对
            if(i!=0&&ch[i-1]!='e'&&ch[i-1]!='E') return false;
        }
        else{
            return false;
        }
    }
    return num;
}

}

19.调整数组顺序使奇数位于偶数前面

解1:暴力搜索,可以不改变相对顺序
解2:快排,但相对顺序可能发生改变

class Solution {
public int[] exchange(int[] nums) {
//快排
if(nums==null||nums.length==0) return nums;
int l = 0,r = nums.length-1;

    while(l<r){
        //r从右往左找奇数
        while(l<r&&nums[r]%2==0) r--;
        //l从左往右找偶数
        while(l<r&&nums[l]%2==1) l++;

        if(l<r){
            int temp = nums[l];
            nums[l] = nums[r];
            nums[r] = temp;
        }
    }
    return nums;
}

}

23.树的子结构

class Solution {
public boolean isSubStructure(TreeNode A, TreeNode B) {
if(B==null||A==null) return false;

    //递归找到值相等的根,进入递归方法判断是否结构相同
    if(A.val==B.val){
        if(sub(A,B)) return true;
    }

    return isSubStructure(A.left,B)||isSubStructure(A.right,B);
}

boolean sub(TreeNode r1,TreeNode r2){
    //如果r2到了最后都顺利 说明匹配正确
    if(r2==null) return true;
    //如果r1空了还没匹配出结果或者值不一样，返回false
    if(r1==null||r1.val!=r2.val) return false;

    //递归 要求左右结构都相同
    return sub(r1.left,r2.left)&&sub(r1.right,r2.right);
}

}

class Solution {
public boolean isSymmetric(TreeNode root) {
if(root==null) return true;
//递归
//判断左右是否对称
return dfs(root.left,root.right);
}

boolean dfs(TreeNode A,TreeNode B){
    if(A==null&&B==null) return true;
    if(A==null||B==null||A.val!=B.val) return false;
    //左等于右 右等于左
    return dfs(A.left,B.right)&&dfs(A.right,B.left);
}

}

26.顺时针打印矩阵

设置标志位,每次打印完判断是否继续

class Solution {
public int[] spiralOrder(int[][] matrix) {
if(matrix==null||matrix.length==0||matrix[0].length==0) return new int[0];
int m = matrix.length;
int n = matrix[0].length;
int[] res = new int[m*n];
int index=0;
//设置四个标志位
int up=0,down=matrix.length-1;
int left=0,right=matrix[0].length-1;
while(left<=right&&up<=down){
//先从左往右
for(int i=left;i<=right;i++){
res[index++]=matrix[up][i];
}
//判断是否继续从上往下
if(++up>down) break;

        //从上而下
        for(int i=up;i<=down;i++){
            res[index++]=matrix[i][right];
        }
        if(--right<left) break;

        //从右往左
        for(int i=right;i>=left;i--){
            res[index++]=matrix[down][i];
        }
        if(--down<up) break;

        //从下往上
        for(int i=down;i>=up;i--){
            res[index++]=matrix[i][left];
        }
        if(++left>right) break;
    }
    return res;
}

}

27.包含min函数的栈

class MinStack {

//利用两个堆栈
//一个正常存，一个存当前最小元素
/** initialize your data structure here. */
Stack<Integer> A;
Stack<Integer> B;
public MinStack() {
    A=new Stack<>();
    B=new Stack<>();
}

public void push(int x) {
    //如果B空或者B当前peek不是最小值，把x加到B中
    if(B.isEmpty()||B.peek()>=x) B.push(x);
    A.push(x);

}

public void pop() {
    //如果B不空且A当前弹出的就是最小值，则B也要弹出
    if(!B.isEmpty()&&B.peek().equals(A.peek())) B.pop();
    A.pop();
}

public int top() {
    return A.peek();
}

public int min() {
    return B.peek();
}

}

//后序遍历： 左右根

//判断这一段是否满足规律
boolean dfs(int[] p,int start,int end){
    if(start>=end) return true;

    int l=start;
    int root = p[end];
    //左边都比根小，找到左右分界点先
    while(p[l]<root) l++;

    int m = l;
    //右边都比根大
    while(p[l]>root) l++;

    //此时如果l没有顺利到最后，说明不满足规则
    if(l!=end) return false;

    //递归判断子问题
    return dfs(p,start,m-1)&&dfs(p,m,end-1);

}

}

    if(data==null||data.length()==0) return null;

    LinkedList<TreeNode> q = new LinkedList<>();
    String[] str = data.substring(1,data.length()-1).split(",");

    TreeNode root = new TreeNode(Integer.valueOf(str[0]));
    q.add(root);
    int index = 1;
    while(!q.isEmpty()){
        TreeNode r = q.poll();
        //先重建左节点
        if(!str[index].equals("null")){
            r.left = new TreeNode(Integer.valueOf(str[index]));
            q.add(r.left);

        }
        index++;
        //再重建右节点
        if(!str[index].equals("null")){
            r.right = new TreeNode(Integer.valueOf(str[index]));
            q.add(r.right);
        }
        index++;
    }
    return root;
}

class Solution {

public int reversePairs(int[] nums) {
//重写归并算法
//一路二分，最后归并
if(nums==null||nums.length<=0) return 0;
temp = new int[nums.length];

    return mergeSort(nums,0,nums.length-1);
}

int mergeSort(int[] nums,int start,int end){
    if(start>=end) return 0;
    int l = start,r = end;
    //一路二分
    int mid = l+(r-l)/2;
    int leftc = mergeSort(nums,start,mid);
    int rightc = mergeSort(nums,mid+1,end);
    //归并
    int cross = merge(nums,start,mid,end);
    return cross+leftc+rightc;
}

int merge(int[] nums,int start,int mid,int end){

    //先存进临时数组
    for(int i=start;i<=end;i++){
        temp[i] = nums[i];
    }

    int l=start,r=mid+1;
    int count = 0;
    for(int index=start;index<=end;index++){
        //如果temp[l]的值比temp[r]大的话，l后面的都比它大，
        //这就存在mid-l+1个逆序对
        if(l>mid){
            nums[index] = temp[r++];
        }
        else if(r>end){
            nums[index] = temp[l++];
        }
        else if(temp[l]>temp[r]){
            nums[index] = temp[r++];
            count+=(mid-l+1);
        }else{
            nums[index] = temp[l++];
        }
    }
    return count;

}

}

    //找sum中1的位置即两个解相异的位
    int index = 0;
    while((sum >> index &1) !=1) index++;

    //按此相异的位分成两组异或
    int res1 = 0;
    for(int n:nums){
        if((n >> index&1)==1) res1^=n;
    }
    int[] res = {res1,sum^res1};
    return res;
}

}

                res.add(temp);
            }
            j++;
        }
    }
    if(res.size()==0) return new int[0][];
    int[][] ans = new int[res.size()][];
    for(int i=0;i<res.size();i++){

        }
        ans[i] = temp;
    }

}

63.滑动窗口的最大值

采用双端队列，队列中的头节点保存的数据比后面的要大。
比如当前假如的数据比队尾的数字大，说明当前这个数字最起码在从现在起到后面的过程中可能是最大值，而之前队尾的数字不可能最大了，所以要删除队尾元素。
此外，还要判断队头的元素是否超过size长度，由于存储的是下标，所以可以计算得到；特别说明，我们在双端队列中保存的数字是传入的向量的下标；

class Solution {
public int[] maxSlidingWindow(int[] nums, int k) {
//LinkedList 用作队列
LinkedList deque = new LinkedList<>();

    //先判断窗口可不可行
    if(k<=0||k>nums.length) return new int[0];
    int[] res = new int[nums.length - k + 1];
    // 未形成窗口
    for(int i = 0; i < k; i++) {
        while(!deque.isEmpty() && deque.peekLast() < nums[i])
            deque.removeLast();
        deque.addLast(nums[i]);
    }
    res[0] = deque.peekFirst();
    // 形成窗口后
    for(int i = k; i < nums.length; i++) {
        if(deque.peekFirst() == nums[i - k])
            deque.removeFirst();
        while(!deque.isEmpty() && deque.peekLast() < nums[i])
            deque.removeLast();
        deque.addLast(nums[i]);
        res[i - k + 1] = deque.peekFirst();
    }
    return res;

}

}

64.队列的最大值

窗口不滑动

class MaxQueue {
//队列常规存取
LinkedList que;
//双端队列维护最大值
LinkedList deq;
public MaxQueue() {
que = new LinkedList();
deq = new LinkedList();
}

public int max_value() {
    //deq队首即为最大值
    return deq.size()==0?-1:deq.peekFirst();
}

public void push_back(int value) {
    //que正常存取
    que.add(value);

    //存入算法与滑动窗口最大值相同
    //只是窗口就是整个队列
    while(!deq.isEmpty()&&deq.peekLast()<value){
        deq.removeLast();
    }
    deq.add(value);
}

public int pop_front() {
    int v = que.size()>0?que.poll():-1;
    if(deq.size()>0&&v==deq.peekFirst()) deq.poll();
    return v;
}

}

65.n个骰子的点数

动态规划 dp[i][j]表示 i个骰子投出j点的次数

class Solution {
public double[] dicesProbability(int n) {
//n个骰子，有n~6n种点数 解的长度为6n-n+1
//总数是6^n次
//概率=dp[i][j]/6^n
//dp[i][j]表示 i个骰子投出j点的次数
double all = Math.pow(6,n);

    //动态规划
    double[] res = new double[5*n+1];
    int[][] dp = new int[n+1][6*n+1];

    //初始化
    dp[0][0] = 1;
    //遍历更新
    //n个骰子
    for(int i=1;i<=n;i++){
        //n~6n种点数
        for(int j=1;j<=6*n;j++){
            //最后一投1~6
            for(int k = 1;k<=Math.min(j,6);k++){
                dp[i][j] += dp[i-1][j-k];
            }
        }
    } 

    for(int i=n;i<=6*n;i++){
        res[i-n] = dp[n][i]/all;
    }
    return res;
}

}

74.二叉树的最近公共祖先

class Solution {
public TreeNode lowestCommonAncestor(TreeNode root, TreeNode p, TreeNode q) {
//递归

    //向下搜索 如果都在左边说明公共节点在左边 如果都在右边说明公共节点在右边

    //递归终止条件：返回null或者找到了p或q
    if(root==null||root==p||root==q) return root;

    //搜索左边有没有p或q
    TreeNode left = lowestCommonAncestor(root.left,p,q);
    //搜索右边有没有p或q
    TreeNode right = lowestCommonAncestor(root.right,p,q);

    //如果left==null 说明都在右边 继续搜索右边
    if(left==null) return right;
    //如果right==null 说明都在左边 继续搜索左边
    if(right==null) return left;

    //左右都不为空 说明pq在root的一左一右 此时root即为公共节点
    return root;

}

}
